# app.py
import streamlit as st
import pandas as pd
from pathlib import Path
import plotly.express as px
import plotly.graph_objects as go
import folium
from streamlit_folium import st_folium
import math

# Use the new pipeline
from pipeline import (
    load_data, make_features, forecast_all, detect_anomalies,
    compute_inventory_policies, optimize_allocation, make_routes,
    plan_dispatch, apply_scenario_simulation, get_dynamic_warehouse_stock,
    DATA_PATH, VIRTUAL_WAREHOUSES, DISTRICT_COORDS
)

st.set_page_config(page_title="Smart Medical Supply Chain", layout="wide")

# ==========================================
# SIDEBAR: DATA & SCENARIO CONTROLS
# ==========================================
st.sidebar.title("Controls")

# 1. Data Source
mode = st.sidebar.radio("Data source", ["Run pipeline now", "Load last outputs"])
uploaded = st.sidebar.file_uploader("Upload CSV (optional)", type=["csv"])

# 2. SCENARIO MANAGEMENT (Advanced Feature)
st.sidebar.markdown("---")-
st.sidebar.subheader("Scenario Management")
sim_type = st.sidebar.selectbox("Active Crisis", ["None", "Pandemic Outbreak", "Natural Disaster (Flood)"])

sim_epicenter = "Ernakulam"  # Default
sim_severity = 0.0

if sim_type != "None":
    sim_epicenter = st.sidebar.selectbox("Epicenter", list(DISTRICT_COORDS.keys()))
    sim_severity = st.sidebar.slider("Severity Index", 0.0, 1.0, 0.5)

# 3. TRAFFIC CONTROL (Interactive)
st.sidebar.markdown("---")
st.sidebar.subheader("🚦 Real-time Logistics")
traffic_val = st.sidebar.slider("Traffic Congestion (%)", 0, 100, 20)

run_btn = st.sidebar.button("Execute / Refresh System")

# ==========================================
# PIPELINE EXECUTION
# ==========================================
data_path = DATA_PATH
if uploaded is not None:
    data_path = uploaded

if mode == "Run pipeline now" and run_btn:
    with st.spinner("Processing Algorithms & Simulating Scenario..."):
        # 1. Load & Simulate
        df_raw = load_data(data_path)
        df_sim = apply_scenario_simulation(df_raw, sim_type, sim_epicenter, sim_severity)

        # 2. ML & Optimization
        df_feat = make_features(df_sim)
        fcst = forecast_all(df_feat)
        alerts = detect_anomalies(df_feat)
        policies = compute_inventory_policies(df_sim, fcst)

        fcst_sum = policies[["District", "Item_Needed", "forecast"]]
        alloc = optimize_allocation(df_sim, fcst_sum)
        routes = make_routes(alloc)
        dispatch = plan_dispatch(policies, traffic_factor=traffic_val)

        # 3. Get Interactive Warehouse Stock
        wh_stock = get_dynamic_warehouse_stock(sim_type, sim_severity)

        # Save to Session
        st.session_state.update(dict(
            df=df_sim, fcst=fcst, alerts=alerts, policies=policies,
            alloc=alloc, routes=routes, dispatch=dispatch, wh_stock=wh_stock
        ))

# ==========================================
# MAIN DASHBOARD (Your Original Layout)
# ==========================================
st.title("Smart Medical Supply Chain Dashboard")

# Filters
c1, c2 = st.columns(2)
district_filter = c1.text_input("Filter: District", "")
item_filter = c2.text_input("Filter: Item", "")


def apply_filters(df):
    if df is None or df.empty: return df
    x = df.copy()
    if district_filter:
        x = x[x["District"].astype(str).str.contains(district_filter, case=False, na=False)]
    if item_filter and "Item_Needed" in x.columns:
        x = x[x["Item_Needed"].astype(str).str.contains(item_filter, case=False, na=False)]
    return x


# TABS RESTORED
tabs = st.tabs(
    ["Forecasts", "Anomalies", "Inventory", "Allocation", "Routes", "Inventory Update / Dispatch Map", "About"])

# TAB 1: FORECASTS
with tabs[0]:
    st.subheader("7-Day Forecasts (Interactive)")
    fcst = apply_filters(st.session_state.get("fcst"))
    if fcst is not None and not fcst.empty:
        fig = px.line(fcst, x="t", y="forecast", color="District", line_group="Item_Needed", title="Demand Projection")
        st.plotly_chart(fig, use_container_width=True)
    else:
        st.info("Run the pipeline to see forecasts.")

# TAB 2: ANOMALIES
with tabs[1]:
    st.subheader("Anomalies & Alerts")
    alerts = apply_filters(st.session_state.get("alerts"))
    if alerts is not None and not alerts.empty:
        st.dataframe(alerts, use_container_width=True)
    else:
        st.success("No anomalies detected.")

# TAB 3: INVENTORY
with tabs[2]:
    st.subheader("Inventory Policy (s, S)")
    pol = apply_filters(st.session_state.get("policies"))
    if pol is not None:
        st.dataframe(pol, use_container_width=True)

# TAB 4: ALLOCATION
with tabs[3]:
    st.subheader("Allocation Plan")
    alloc = apply_filters(st.session_state.get("alloc"))
    if alloc is not None:
        st.dataframe(alloc, use_container_width=True)

# TAB 5: ROUTES
with tabs[4]:
    st.subheader("Optimized Logistics Routes")
    routes = st.session_state.get("routes")
    if routes is not None:
        st.dataframe(routes, use_container_width=True)
        st.caption(f"Traffic Congestion Level: {traffic_val}%")

# TAB 6: WAREHOUSE & MAP (The Requested Interactive Hub Section)
with tabs[5]:
    st.subheader("Warehouse & Emergency Dispatch")

    # -----------------------------
    # 1. INTERACTIVE WAREHOUSE HUBS
    # -----------------------------
    st.markdown("###  Regional Warehouse Status")
    wh_stock = st.session_state.get("wh_stock")

    if wh_stock:
        wc1, wc2, wc3 = st.columns(3)
        hubs = ["Central Hub", "North Hub", "South Hub"]
        cols = [wc1, wc2, wc3]

        for i, hub in enumerate(hubs):
            with cols[i]:
                st.markdown(f"**{hub}**")
                stock_data = wh_stock.get(hub, {})

                # Interactive Metrics per Item
                for item, qty in stock_data.items():
                    # Color logic: Red if depleted by crisis
                    delta_color = "normal"
                    if sim_type != "None" and qty < 500: delta_color = "inverse"

                    st.metric(label=item, value=qty, delta="-CRITICAL" if qty < 300 else "OK", delta_color=delta_color)
                st.divider()
    else:
        st.info("Execute pipeline to load warehouse status.")

    # -----------------------------
    # 2. INTERACTIVE DISPATCH MAP
    # -----------------------------
    st.markdown("###  Live Dispatch Tracking")
    dispatch = st.session_state.get("dispatch")

    m = folium.Map(location=[10.5, 76.2], zoom_start=7)

    # Plot Hubs
    for _, w in VIRTUAL_WAREHOUSES.iterrows():
        folium.Marker(
            [w["lat"], w["lon"]], tooltip=w["Warehouse"],
            icon=folium.Icon(color="blue", icon="home", prefix="fa")
        ).add_to(m)

    # Plot Routes (Red if Traffic High)
    if dispatch is not None and not dispatch.empty:
        line_color = "red" if traffic_val > 50 else "green"

        for _, row in dispatch.iterrows():
            folium.PolyLine(
                [(row["w_lat"], row["w_lon"]), (row["d_lat"], row["d_lon"])],
                color=line_color, weight=2.5, opacity=0.8,
                tooltip=f"ETA: {int(row['ETA_min'])} mins"
            ).add_to(m)

            folium.Marker(
                [row["d_lat"], row["d_lon"]],
                tooltip=f"{row['District']} ({row['Item_Needed']})",
                icon=folium.Icon(color="orange", icon="truck", prefix="fa")
            ).add_to(m)

    st_folium(m, width=1000, height=600)

# TAB 7: ABOUT
with tabs[6]:
    st.markdown(""" 
    ### System Capabilities 
    - **Scenario Management:** Simulates Pandemics and Floods with adjustable severity. 
    - **Dynamic Warehousing:** Warehouse stock depletes in real-time based on crisis impact. 
    - **Live Logistics:** ETAs and Map Routes react to traffic congestion levels. 
    """)
