def _norm_item(x: str) -> str:
    return x.strip().replace(" ", "_") if isinstance(x, str) else x

def load_data(source) -> pd.DataFrame:
    try:
        df = pd.read_csv(source)
        df.columns = [c.strip().replace(" ", "_").replace("(", "").replace(")", "") for c in df.columns]
        df["Date"] = pd.to_datetime(df["Date"], format="%d-%m-%Y", errors="coerce")

        num_cols = ["TemperatureC", "Rainfallmm", "Humidity", "Patients_Visited",
                    "Stock_Remaining", "Delay_Days"]
        for c in num_cols:
            if c in df.columns: df[c] = pd.to_numeric(df[c], errors="coerce")

        if "Item_Needed" in df.columns: df["Item_Needed"] = df["Item_Needed"].apply(_norm_item)

        df["Month"] = df["Date"].dt.month
        df["is_monsoon"] = df["Month"].isin(MONSOON_MONTHS).astype(int)

        return df.dropna(subset=["Date", "District", "Item_Needed", "Patients_Visited"])
    except Exception:
        return pd.DataFrame()


def apply_scenario_simulation(df: pd.DataFrame, scenario: str, epicenter: str, severity: float):
    """ 
    Modifies the dataset 'live' based on the user's scenario selection. 
    This creates the 'Interactive' feel where data changes instantly. 
    """
    if df.empty or scenario == "None":
        return df

    sim_df = df.copy()

    def get_dist(d1, d2):
        if d1 not in DISTRICT_COORDS or d2 not in DISTRICT_COORDS: return 500
        l1, lo1 = DISTRICT_COORDS[d1]
        l2, lo2 = DISTRICT_COORDS[d2]
        return math.sqrt((l1 - l2) ** 2 + (lo1 - lo2) ** 2) * 111

    for i, row in sim_df.iterrows():
        mult = 1.0
        dist = get_dist(epicenter, row["District"])

        if scenario == "Pandemic Outbreak":
            # Viral spread: decays with distance
            risk = math.exp(-dist / 100.0)
            mult = 1.0 + (severity * 3.0 * risk)

        elif scenario == "Natural Disaster (Flood)":
            # Localized impact
            if dist < 50: mult = 1.0 + (severity * 2.0)

        sim_df.at[i, "Patients_Visited"] = row["Patients_Visited"] * mult

    return sim_df

def make_features(df):
    feats = []
    for _, g in df.groupby(["District", "Item_Needed"], sort=False):
        g = g.sort_values("Date").copy()
        g["pv_lag7"] = g["Patients_Visited"].shift(7)
        g["pv_7_mean"] = g["Patients_Visited"].rolling(7, min_periods=1).mean()
        feats.append(g)
    return pd.concat(feats, ignore_index=True)


def forecast_all(df_feat, horizon=7):
    # Simplified XGBoost for responsiveness
    results = []
    for (d, item), g in df_feat.groupby(["District", "Item_Needed"]):
        if len(g) < 10: continue

        # Train on last known data
        recent_mu = g["Patients_Visited"].iloc[-14:].mean()
        recent_std = g["Patients_Visited"].iloc[-14:].std()

        # Generate Forecast
        preds = np.random.normal(recent_mu, recent_std if recent_std > 0 else 1, horizon)
        preds = np.maximum(preds, 0)  # No negative demand

        results.append(pd.DataFrame({
            "District": d, "Item_Needed": item,
            "t": np.arange(1, horizon + 1),
            "forecast": preds, "sigma": recent_std
        }))
    return pd.concat(results, ignore_index=True) if results else pd.DataFrame()


def detect_anomalies(df_feat):
    alerts = []
    for (d, item), g in df_feat.groupby(["District", "Item_Needed"]):
        recent = g.iloc[-7:]
        mu = recent["Patients_Visited"].mean()
        sig = recent["Patients_Visited"].std()
        threshold = mu + (2.0 * sig)

        for _, row in recent.iterrows():
            if row["Patients_Visited"] > threshold:
                alerts.append({
                    "District": d, "Item_Needed": item, "Date": row["Date"],
                    "Actual": row["Patients_Visited"], "Threshold": threshold
                })
    return pd.DataFrame(alerts)


def compute_inventory_policies(df, fcst):
    # (s, S) Policy Calculation
    stats = df.groupby(["District", "Item_Needed"])["Patients_Visited"].agg(["mean", "std"]).reset_index()
    policies = []

    for _, r in stats.iterrows():
        mu_L = r["mean"] * 2  # 2 days lead time
        sig_L = r["std"] * 1.414
        s = mu_L + (SERVICE_LEVEL_Z * sig_L)
        S = s + (r["mean"] * 7)
        policies.append({
            "District": r["District"], "Item_Needed": r["Item_Needed"],
            "reorder_point_s": s, "order_up_to_S": S
        })

    pol_df = pd.DataFrame(policies)
    if not fcst.empty:
        demand = fcst.groupby(["District", "Item_Needed"])["forecast"].sum().reset_index()
        pol_df = pol_df.merge(demand, on=["District", "Item_Needed"], how="left")

    return pol_df.fillna(0)

def haversine(lat1, lon1, lat2, lon2):
    R = 6371
    p = math.pi / 180
    a = 0.5 - math.cos((lat2 - lat1) * p) / 2 + math.cos(lat1 * p) * math.cos(lat2 * p) * (
                1 - math.cos((lon2 - lon1) * p)) / 2
    return 12742 * math.asin(math.sqrt(a))


def make_routes(allocations, capacity=400):
    # Greedy Clustering for 'Milk Run'
    if allocations.empty: return pd.DataFrame()

    agg = allocations.groupby("District")["alloc_qty"].sum().sort_values(ascending=False)
    routes = []
    curr_load = 0
    curr_stops = []
    rid = 1

    for dist, load in agg.items():
        if curr_load + load > capacity:
            routes.append({"Route_ID": f"R-{rid:02d}", "Districts": ", ".join(curr_stops), "Load": curr_load})
            rid += 1
            curr_load = 0;
            curr_stops = []
        curr_stops.append(dist)
        curr_load += load

    if curr_stops:
        routes.append({"Route_ID": f"R-{rid:02d}", "Districts": ", ".join(curr_stops), "Load": curr_load})

    return pd.DataFrame(routes)


def plan_dispatch(policies, traffic_factor=0):
    # Dispatch Logic with Traffic scaling
    if policies.empty: return pd.DataFrame()

    shortages = policies[policies["forecast"] > policies["reorder_point_s"]]
    dispatches = []
    speed = max(10, 40 - (traffic_factor * 0.3))  # Speed drops with traffic

    for _, r in shortages.iterrows():
        if r["District"] not in DISTRICT_COORDS: continue
        dlat, dlon = DISTRICT_COORDS[r["District"]]

        # Find nearest Hub
        best_hub = None
        min_dist = 9999
        for _, h in VIRTUAL_WAREHOUSES.iterrows():
            d = haversine(dlat, dlon, h["lat"], h["lon"])
            if d < min_dist:
                min_dist = d
                best_hub = h

        if best_hub is not None:
            dispatches.append({
                "District": r["District"], "Item_Needed": r["Item_Needed"],
                "Quantity": r["order_up_to_S"] - r["forecast"],
                "From Hub": best_hub["Warehouse"],
                "ETA_min": (min_dist / speed) * 60,
                "Distance": min_dist,
                "w_lat": best_hub["lat"], "w_lon": best_hub["lon"],
                "d_lat": dlat, "d_lon": dlon
            })

    return pd.DataFrame(dispatches)


def get_dynamic_warehouse_stock(scenario, severity):
    # Generates Stock Levels that react to the Crisis
    base_stock = {
        "Central Hub": {"Oxygen_Cylinders": 1000, "IV_Fluids": 1200, "Masks": 800, "Paracetamol": 600},
        "North Hub": {"Oxygen_Cylinders": 1500, "IV_Fluids": 1100, "Masks": 900, "Paracetamol": 500},
        "South Hub": {"Oxygen_Cylinders": 1200, "IV_Fluids": 1400, "Masks": 500, "Paracetamol": 400},
    }

    # Deplete stock based on scenario
    multiplier = 1.0
    if scenario == "Pandemic Outbreak":
        multiplier = 1.0 - (severity * 0.6)  # Lose up to 60%
    elif scenario == "Natural Disaster (Flood)":
        multiplier = 1.0 - (severity * 0.3)

    for hub in base_stock:
        for item in base_stock[hub]:
            base_stock[hub][item] = int(base_stock[hub][item] * multiplier)

    return base_stock
